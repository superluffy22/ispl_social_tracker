<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useMemo } = React;

        const { 

            BarChart, Bar, XAxis, YAxis, CartesianGrid, 

            Tooltip, ResponsiveContainer, LineChart, Line, Legend 

        } = Recharts;


        // --- Constants ---

        const SEASON_TARGETS = {

            "Off-Season":  { static: 60,  video: 30  },

            "Pre-Season":  { static: 175, video: 45  },

            "In-Season":   { static: 250, video: 100 },

        };

        const MONTHS = ["January","February","March","April","May","June","July","August","September","October","November","December"];

        const SEASON_TYPES = ["Off-Season", "Pre-Season", "In-Season"];

        const STORAGE_KEY = "smpt_entries_v1";


        // --- Helpers ---

        const calcEntry = (e) => {

            const st = Number(e.staticTarget) || 0;

            const sp = Number(e.staticPosted) || 0;

            const vt = Number(e.videoTarget) || 0;

            const vp = Number(e.videoPosted) || 0;

            const xt = Number(e.storiesTarget) || 0;

            const xp = Number(e.storiesPosted) || 0;

            const totalTarget = st + vt + xt;

            const totalPosted = sp + vp + xp;

            return {

                staticGap: st - sp,

                videoGap: vt - vp,

                storiesGap: xt - xp,

                totalTarget,

                totalPosted,

                totalGap: totalTarget - totalPosted,

                perf: totalTarget > 0 ? Math.round((totalPosted / totalTarget) * 100) : 0,

            };

        };


        // --- Components ---

        const ProgressBar = ({ pct, color }) => {

            const c = color || (pct >= 100 ? "#22c55e" : pct >= 70 ? "#facc15" : "#ef4444");

            return (

                <div className="bg-white/10 rounded-full h-1.5 w-full overflow-hidden">

                    <div 

                        className="h-full rounded-full transition-all duration-500" 

                        style={{ width: `${Math.min(pct, 100)}%`, backgroundColor: c }}

                    />

                </div>

            );

        };


        const PerfBadge = ({ pct }) => {

            const c = pct >= 100 ? "#22c55e" : pct >= 70 ? "#facc15" : "#ef4444";

            return (

                <span className="px-2 py-0.5 rounded-full text-[10px] font-bold border" 

                      style={{ backgroundColor: `${c}22`, color: c, borderColor: `${c}44` }}>

                    {pct}%

                </span>

            );

        };


        const App = () => {

            const [entries, setEntries] = useState(() => {

                try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; } catch { return []; }

            });

            const [tab, setTab] = useState("dashboard");

            const [editing, setEditing] = useState(null);

            const [filterSeason, setFilterSeason] = useState("All");


            useEffect(() => {

                localStorage.setItem(STORAGE_KEY, JSON.stringify(entries));

            }, [entries]);


            const totals = useMemo(() => entries.reduce((a, e) => {

                const c = calcEntry(e);

                return { 

                    target: a.target + c.totalTarget, 

                    posted: a.posted + c.totalPosted,

                    staticT: a.staticT + (Number(e.staticTarget)||0),

                    staticP: a.staticP + (Number(e.staticPosted)||0),

                    videoT: a.videoT + (Number(e.videoTarget)||0),

                    videoP: a.videoP + (Number(e.videoPosted)||0),

                    storyT: a.storyT + (Number(e.storiesTarget)||0),

                    storyP: a.storyP + (Number(e.storiesPosted)||0)

                };

            }, { target: 0, posted: 0, staticT: 0, staticP: 0, videoT: 0, videoP: 0, storyT: 0, storyP: 0 }), [entries]);


            const overallPerf = totals.target > 0 ? Math.round((totals.posted / totals.target) * 100) : 0;


            const chartData = entries

                .filter(e => filterSeason === "All" || e.season === filterSeason)

                .map(e => ({

                    month: e.month.slice(0, 3),

                    staticTarget: Number(e.staticTarget) || 0,

                    staticPosted: Number(e.staticPosted) || 0,

                    videoTarget: Number(e.videoTarget) || 0,

                    videoPosted: Number(e.videoPosted) || 0,

                    perf: calcEntry(e).perf

                }));


            return (

                <div className="min-h-screen pb-10">

                    {/* Header */}

                    <nav className="sticky top-0 z-50 backdrop-blur-md bg-black/40 border-b border-white/10 px-6 h-16 flex items-center justify-between no-print">

                        <div className="flex items-center gap-3">

                            <div className="w-8 h-8 rounded-lg bg-indigo-600 flex items-center justify-center shadow-lg shadow-indigo-500/20">📊</div>

                            <span className="font-extrabold text-lg tracking-tight">Social Tracker</span>

                        </div>

                        <div className="flex gap-2">

                            {["dashboard", "add", "table", "chart"].map(t => (

                                <button 

                                    key={t}

                                    onClick={() => setTab(t)}

                                    className={`px-4 py-1.5 rounded-lg text-xs font-semibold transition-all ${tab === t ? 'bg-indigo-600 text-white' : 'bg-white/5 text-gray-400 hover:bg-white/10'}`}

                                >

                                    {t.charAt(0).toUpperCase() + t.slice(1)}

                                </button>

                            ))}

                        </div>

                        <button onClick={() => window.print()} className="bg-white/5 border border-white/10 px-3 py-1.5 rounded-lg text-xs font-bold text-gray-300 hover:bg-white/10">

                            Export PDF

                        </button>

                    </nav>


                    <main className="max-w-6xl mx-auto p-6 md:p-10">

                        {tab === 'dashboard' && (

                            <div className="space-y-8 animate-in fade-in duration-500">

                                <div>

                                    <h1 className="text-3xl font-black tracking-tight">Performance</h1>

                                    <p className="text-gray-500 text-sm">Aggregated content metrics across all platforms.</p>

                                </div>


                                {/* Banner */}

                                <div className={`p-8 rounded-2xl border flex flex-col md:flex-row justify-between items-center gap-8 ${

                                    overallPerf >= 100 ? 'bg-green-500/5 border-green-500/20' : overallPerf >= 70 ? 'bg-yellow-500/5 border-yellow-500/20' : 'bg-red-500/5 border-red-500/20'

                                }`}>

                                    <div className="text-center md:text-left">

                                        <p className="text-[10px] uppercase tracking-widest font-bold text-gray-500 mb-1">Overall Completion</p>

                                        <h2 className={`text-6xl font-black ${overallPerf >= 100 ? 'text-green-500' : overallPerf >= 70 ? 'text-yellow-500' : 'text-red-500'}`}>

                                            {overallPerf}%

                                        </h2>

                                        <p className="text-gray-400 text-sm mt-2 font-medium">{totals.posted} posts / {totals.target} target</p>

                                    </div>

                                    <div className="w-full max-w-sm">

                                        <ProgressBar pct={overallPerf} />

                                        <div className="flex justify-between mt-3 text-[10px] font-bold text-gray-600 uppercase tracking-tighter">

                                            <span>Start</span>

                                            <span>Goal Achieved</span>

                                        </div>

                                    </div>

                                </div>


                                {/* Grid Stats */}

                                <div className="grid grid-cols-2 md:grid-cols-4 gap-4">

                                    {[

                                        { label: "Static Posts", val: totals.staticP, sub: `Target: ${totals.staticT}`, color: "text-indigo-400" },

                                        { label: "Video/Reels", val: totals.videoP, sub: `Target: ${totals.videoT}`, color: "text-purple-400" },

                                        { label: "Total Gap", val: totals.target - totals.posted, sub: "Remaining", color: "text-red-400" },

                                        { label: "Months", val: entries.length, sub: "Tracked", color: "text-white" }

                                    ].map((s, i) => (

                                        <div key={i} className="bg-white/5 border border-white/5 p-6 rounded-xl">

                                            <p className="text-[10px] font-bold text-gray-500 uppercase mb-2">{s.label}</p>

                                            <p className={`text-3xl font-bold ${s.color}`}>{s.val}</p>

                                            <p className="text-xs text-gray-600 mt-1">{s.sub}</p>

                                        </div>

                                    ))}

                                </div>

                            </div>

                        )}


                        {tab === 'add' && (

                            <div className="max-w-2xl mx-auto bg-white/5 p-8 rounded-2xl border border-white/10">

                                <h2 className="text-xl font-bold mb-6">{editing ? "Edit Entry" : "New Monthly Entry"}</h2>

                                <form onSubmit={(e) => {

                                    e.preventDefault();

                                    const fd = new FormData(e.target);

                                    const data = Object.fromEntries(fd.entries());

                                    const newEntry = { ...data, id: editing?.id || Date.now() };

                                    setEntries(prev => editing ? prev.map(x => x.id === editing.id ? newEntry : x) : [...prev, newEntry]);

                                    setEditing(null);

                                    setTab("table");

                                }} className="grid grid-cols-2 gap-4">

                                    <div className="col-span-1">

                                        <label className="text-[10px] font-bold text-gray-500 uppercase block mb-1">Month</label>

                                        <select name="month" defaultValue={editing?.month || MONTHS[new Date().getMonth()]} className="w-full bg-white/10 border border-white/10 rounded-lg p-2 text-sm outline-none">

                                            {MONTHS.map(m => <option key={m} value={m}>{m}</option>)}

                                        </select>

                                    </div>

                                    <div className="col-span-1">

                                        <label className="text-[10px] font-bold text-gray-500 uppercase block mb-1">Season</label>

                                        <select name="season" defaultValue={editing?.season || "Off-Season"} className="w-full bg-white/10 border border-white/10 rounded-lg p-2 text-sm outline-none">

                                            {SEASON_TYPES.map(s => <option key={s} value={s}>{s}</option>)}

                                        </select>

                                    </div>

                                    {["staticTarget", "staticPosted", "videoTarget", "videoPosted", "storiesTarget", "storiesPosted"].map(field => (

                                        <div key={field}>

                                            <label className="text-[10px] font-bold text-gray-500 uppercase block mb-1">{field.replace(/([A-Z])/g, ' $1')}</label>

                                            <input name={field} type="number" defaultValue={editing?.[field] || ""} className="w-full bg-white/10 border border-white/10 rounded-lg p-2 text-sm outline-none" required />

                                        </div>

                                    ))}

                                    <div className="col-span-2 pt-4 flex gap-2">

                                        <button type="submit" className="bg-indigo-600 px-6 py-2 rounded-lg font-bold text-sm">Save Entry</button>

                                        {editing && <button type="button" onClick={() => setEditing(null)} className="bg-white/10 px-6 py-2 rounded-lg font-bold text-sm text-gray-400">Cancel</button>}

                                    </div>

                                </form>

                            </div>

                        )}


                        {tab === 'table' && (

                            <div className="overflow-hidden bg-white/5 border border-white/10 rounded-xl">

                                <div className="p-4 border-b border-white/10 flex justify-between items-center overflow-x-auto">

                                    <h3 className="font-bold text-sm">Entry History</h3>

                                    <div className="flex gap-2">

                                        {["All", ...SEASON_TYPES].map(s => (

                                            <button key={s} onClick={() => setFilterSeason(s)} className={`px-3 py-1 rounded text-[10px] font-bold ${filterSeason === s ? 'bg-indigo-600' : 'bg-white/10'}`}>{s}</button>

                                        ))}

                                    </div>

                                </div>

                                <div className="overflow-x-auto custom-scrollbar">

                                    <table className="w-full text-left text-xs">

                                        <thead>

                                            <tr className="bg-white/5 text-gray-500 uppercase text-[9px] font-black tracking-widest">

                                                <th className="p-4">Month</th>

                                                <th className="p-4">Static (P/T)</th>

                                                <th className="p-4">Video (P/T)</th>

                                                <th className="p-4">Performance</th>

                                                <th className="p-4">Actions</th>

                                            </tr>

                                        </thead>

                                        <tbody>

                                            {entries.filter(e => filterSeason === "All" || e.season === filterSeason).map(e => {

                                                const c = calcEntry(e);

                                                return (

                                                    <tr key={e.id} className="border-t border-white/5 hover:bg-white/[0.02]">

                                                        <td className="p-4 font-bold">{e.month} <span className="block text-[10px] text-indigo-400 font-medium">{e.season}</span></td>

                                                        <td className="p-4 text-gray-400">{e.staticPosted} / <span className="text-gray-600">{e.staticTarget}</span></td>

                                                        <td className="p-4 text-gray-400">{e.videoPosted} / <span className="text-gray-600">{e.videoTarget}</span></td>

                                                        <td className="p-4"><PerfBadge pct={c.perf} /></td>

                                                        <td className="p-4 space-x-2">

                                                            <button onClick={() => { setEditing(e); setTab("add"); }} className="text-indigo-400 font-bold">Edit</button>

                                                            <button onClick={() => { if(confirm("Delete?")) setEntries(prev => prev.filter(x => x.id !== e.id)) }} className="text-red-400 font-bold">Del</button>

                                                        </td>

                                                    </tr>

                                                );

                                            })}

                                        </tbody>

                                    </table>

                                </div>

                            </div>

                        )}


                        {tab === 'chart' && (

                            <div className="grid gap-6">

                                <div className="bg-white/5 p-6 rounded-2xl border border-white/10 h-[350px]">

                                    <h3 className="text-[10px] font-bold text-gray-500 uppercase mb-6">Content Delivery Trend</h3>

                                    <ResponsiveContainer width="100%" height="100%">

                                        <BarChart data={chartData}>

                                            <CartesianGrid strokeDasharray="3 3" stroke="#222" vertical={false} />
